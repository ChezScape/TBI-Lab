import fs from "node:fs";
import assert from "node:assert/strict";

const read = file => fs.readFileSync(file, "utf8");

const app = read("./app.js");
const bootstrap = read("./bootstrap.js");
const index = read("./index.html");
const config = read("./config/appConfig.js");
const actionAudit = read("./src/ui/actionAuditBridge.js");
const nativeGuard = read("./src/ui/nativeControlGuard.js");
const importBridge = read("./src/ui/nativeImportHardBridge.js");
const downloadBridge = read("./src/ui/universalDownloadBridge.js");
const searchGuard = read("./src/ui/historySearchFocusGuard.js");

assert.match(config, /version:\s*"v4\.10y"/);
assert.match(app, /APP ENTRY v4\.10y/);
assert.match(bootstrap, /BOOTSTRAP v4\.10y/);

assert.match(index, /src="\.\/src\/ui\/universalDownloadBridge\.js"/);
assert.match(index, /src="\.\/src\/ui\/historySearchFocusGuard\.js"/);
assert.match(index, /type="module" src="\.\/app\.js"/);
assert.ok(index.indexOf("universalDownloadBridge.js") < index.indexOf("app.js"), "download bridge should load before app.js");
assert.ok(index.indexOf("historySearchFocusGuard.js") < index.indexOf("app.js"), "history search guard should load before app.js");

assert.match(bootstrap, /bindNativeControlGuard\(\)/);
assert.match(bootstrap, /nativeImportHardBridge\.js/);
assert.match(bootstrap, /finalControlPolishBridge\.js/);
assert.match(bootstrap, /actionAuditBridge\.js/);
assert.match(bootstrap, /bindStaticControlBridge\(\(\) => render\(\)\)/);
assert.match(bootstrap, /bindLiveInteractionBridge\(\(\) => render\(\)\)/);

assert.match(actionAudit, /const VERSION = "v4\.10y";/);
assert.match(nativeGuard, /const VERSION = "v4\.10y";/);
assert.match(importBridge, /const VERSION = "v4\.10y";/);
assert.match(downloadBridge, /const VERSION = "v4\.10y";/);
assert.match(searchGuard, /const VERSION = "v4\.10y";/);

assert.doesNotMatch(index, /apply-v4\.10/);
assert.equal(fs.existsSync("./patch"), false, "patch folder should not be present in clean checkpoint");
assert.equal(fs.readdirSync(".").some(name => /^apply-v4\.10.*\.ps1$/.test(name)), false, "old apply scripts should not be in root");
assert.equal(fs.readdirSync(".").some(name => /^README_DROP_IN_PATCH/.test(name)), false, "old drop-in patch READMEs should not be in root");

console.log("current-v4.11a-checkpoint.test.mjs passed");
